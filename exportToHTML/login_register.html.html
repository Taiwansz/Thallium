    <script>
        const toggleRegister = document.getElementById('toggle-register');
        const toggleLogin = document.getElementById('toggle-login');
        const loginForm = document.getElementById('login-form').parentElement;
        const registerForm = document.getElementById('register-form').parentElement;

        // Exibir o formulário de registro e ocultar o de login
        toggleRegister.addEventListener('click', (e) => {
            e.preventDefault(); // Previne o comportamento padrão do link
            loginForm.style.display = 'none'; // Oculta o formulário de login
            registerForm.style.display = 'block'; // Exibe o formulário de cadastro
        });

        // Exibir o formulário de login e ocultar o de registro
        toggleLogin.addEventListener('click', (e) => {
            e.preventDefault(); // Previne o comportamento padrão do link
            loginForm.style.display = 'block'; // Exibe o formulário de login
            registerForm.style.display = 'none'; // Oculta o formulário de cadastro
        });
    </script>
